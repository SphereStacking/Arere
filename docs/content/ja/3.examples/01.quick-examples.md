# クイックサンプル

コピペですぐに使えるアクションのサンプル集です。

## 基本的なアクション

### Hello World

```typescript
// ~/.arere/hello.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'hello',
  description: 'Hello World を表示',
  async run({ tui }) {
    tui.output.log('Hello, World!')
  }
})
```

### ユーザーに挨拶

```typescript
// ~/.arere/greet.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'greet',
  description: '挨拶する',
  async run({ tui }) {
    const name = await tui.prompt.text('お名前は？')
    tui.output.success(`こんにちは、${name}さん！`)
  }
})
```

### 環境選択

```typescript
// ~/.arere/env-select.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'env-select',
  description: '環境を選択',
  async run({ tui }) {
    const env = await tui.prompt.select('環境を選択:', [
      { label: '開発', value: 'development' },
      { label: 'ステージング', value: 'staging' },
      { label: '本番', value: 'production' },
    ])
    tui.output.info(`選択: ${env}`)
  }
})
```

## Git 操作

### Git Status

```typescript
// ~/.arere/git-status.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'git-status',
  description: 'Git のステータスを表示',
  category: 'git',
  async run({ $, tui }) {
    const { stdout } = await $`git status --short`
    tui.output.log(stdout || '変更はありません')
  }
})
```

### Git Commit

```typescript
// ~/.arere/git-commit.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'git-commit',
  description: 'Git コミットを作成',
  category: 'git',
  async run({ $, tui }) {
    const { stdout } = await $`git diff --cached --name-only`
    const files = stdout.split('\n').filter(Boolean)

    if (files.length === 0) {
      tui.output.warn('ステージされたファイルがありません')
      return
    }

    tui.output.info('変更されたファイル:')
    files.forEach(f => tui.output.log(`  - ${f}`))

    const message = await tui.prompt.text('コミットメッセージ:')
    await $`git commit -m ${message}`
    tui.output.success('コミット完了')
  }
})
```

### Git Switch

```typescript
// ~/.arere/git-switch.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'git-switch',
  description: 'ブランチを切り替え',
  category: 'git',
  async run({ $, tui }) {
    const { stdout } = await $`git branch --format=%(refname:short)`
    const branches = stdout.split('\n').filter(Boolean)

    const branch = await tui.prompt.select('ブランチを選択:', branches)
    await $`git switch ${branch}`
    tui.output.success(`${branch} に切り替えました`)
  }
})
```

## Docker 操作

### Docker コンテナ一覧

```typescript
// ~/.arere/docker-ps.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'docker-ps',
  description: 'Docker コンテナ一覧',
  category: 'docker',
  async run({ $, tui }) {
    const { stdout } = await $`docker ps --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"`
    tui.output.log(stdout)
  }
})
```

### Docker Logs

```typescript
// ~/.arere/docker-logs.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'docker-logs',
  description: 'Docker コンテナのログを表示',
  category: 'docker',
  async run({ $, tui }) {
    const { stdout } = await $`docker ps --format {{.Names}}`
    const containers = stdout.split('\n').filter(Boolean)

    if (containers.length === 0) {
      tui.output.warn('実行中のコンテナがありません')
      return
    }

    const container = await tui.prompt.select('コンテナを選択:', containers)
    const { stdout: logs } = await $`docker logs --tail 50 ${container}`
    tui.output.log(logs)
  }
})
```

## npm 操作

### npm run スクリプト選択

```typescript
// ~/.arere/npm-run.ts
import { defineAction } from 'arere'
import { readFileSync } from 'node:fs'

export default defineAction({
  name: 'npm-run',
  description: 'npm スクリプトを実行',
  category: 'npm',
  async run({ $, tui }) {
    try {
      const pkg = JSON.parse(readFileSync('package.json', 'utf-8'))
      const scripts = Object.keys(pkg.scripts || {})

      if (scripts.length === 0) {
        tui.output.warn('スクリプトがありません')
        return
      }

      const script = await tui.prompt.select('スクリプトを選択:', scripts)
      await $`npm run ${script}`
    } catch {
      tui.output.error('package.json が見つかりません')
    }
  }
})
```
