# Quick Examples

A collection of actions you can copy and paste to use immediately.

## Basic Actions

### Hello World

```typescript
// ~/.arere/hello.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'hello',
  description: 'Display Hello World',
  async run({ tui }) {
    tui.output.log('Hello, World!')
  }
})
```

### Greet User

```typescript
// ~/.arere/greet.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'greet',
  description: 'Greet someone',
  async run({ tui }) {
    const name = await tui.prompt.text('What is your name?')
    tui.output.success(`Hello, ${name}!`)
  }
})
```

### Environment Selection

```typescript
// ~/.arere/env-select.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'env-select',
  description: 'Select environment',
  async run({ tui }) {

  }
})
```

## Git Operations

### Git Status

```typescript
// ~/.arere/git-status.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'git-status',
  description: 'Display Git status',
  category: 'git',
  async run({ $, tui }) {
    const { stdout } = await $`git status --short`
    tui.output.log(stdout || 'No changes')
  }
})
```

### Git Commit

```typescript
// ~/.arere/git-commit.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'git-commit',
  description: 'Create a Git commit',
  category: 'git',
  async run({ $, tui }) {
    const { stdout } = await $`git diff --cached --name-only`
    const files = stdout.split('\n').filter(Boolean)

    if (files.length === 0) {
      tui.output.warn('No staged files')
      return
    }

    tui.output.info('Changed files:')
    files.forEach(f => tui.output.log(`  - ${f}`))

    const message = await tui.prompt.text('Commit message:')
    await $`git commit -m ${message}`
    tui.output.success('Commit completed')
  }
})
```

### Git Switch

```typescript
// ~/.arere/git-switch.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'git-switch',
  description: 'Switch branches',
  category: 'git',
  async run({ $, tui }) {
    const { stdout } = await $`git branch --format=%(refname:short)`
    const branches = stdout.split('\n').filter(Boolean)

    const branch = await tui.prompt.select('Select branch:', branches)
    await $`git switch ${branch}`
    tui.output.success(`Switched to ${branch}`)
  }
})
```

## Docker Operations

### Docker Container List

```typescript
// ~/.arere/docker-ps.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'docker-ps',
  description: 'List Docker containers',
  category: 'docker',
  async run({ $, tui }) {
    const { stdout } = await $`docker ps --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"`
    tui.output.log(stdout)
  }
})
```

### Docker Logs

```typescript
// ~/.arere/docker-logs.ts
import { defineAction } from 'arere'

export default defineAction({
  name: 'docker-logs',
  description: 'Display Docker container logs',
  category: 'docker',
  async run({ $, tui }) {
    const { stdout } = await $`docker ps --format {{.Names}}`
    const containers = stdout.split('\n').filter(Boolean)

    if (containers.length === 0) {
      tui.output.warn('No running containers')
      return
    }

    const container = await tui.prompt.select('Select container:', containers)
    const { stdout: logs } = await $`docker logs --tail 50 ${container}`
    tui.output.log(logs)
  }
})
```

## npm Operations

### npm run Script Selection

```typescript
// ~/.arere/npm-run.ts
import { defineAction } from 'arere'
import { readFileSync } from 'node:fs'

export default defineAction({
  name: 'npm-run',
  description: 'Run npm script',
  category: 'npm',
  async run({ $, tui }) {
    try {
      const pkg = JSON.parse(readFileSync('package.json', 'utf-8'))
      const scripts = Object.keys(pkg.scripts || {})

      if (scripts.length === 0) {
        tui.output.warn('No scripts found')
        return
      }

      const script = await tui.prompt.select('Select script:', scripts)
      await $`npm run ${script}`
    } catch {
      tui.output.error('package.json not found')
    }
  }
})
```
